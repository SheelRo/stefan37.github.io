using GTANetworkAPI;
namespace Server.Systems {
public class PlayerSystem : Script {
[ServerEvent(Event.PlayerConnected)]
public void OnPlayerConnected(Player player) {
player.SendChatMessage("Welcome!");
player.Position = new Vector3(
-425.52, 1123.46, 325.85);
NAPI.Player.SpawnPlayer(player,
player.Position);
Database.LoadAccount(player);
}
[Command("veh")]
public void SpawnVehicle(Player p,
string model) {
var veh = NAPI.Vehicle.CreateVehicle(
NAPI.Util.GetHashKey(model),
p.Position, p.Rotation.Z,
0, 0, p.Dimension);
p.SetIntoVehicle(veh, -1);
}
}
}
